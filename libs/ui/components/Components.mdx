## SegmentedControl

import { SegmentedControl, SegmentedControlOption } from "./SegmentedControl"

<Playground>
  <State initial={{ value: "all" }}>
    {({ state, setState }) => (
      <SegmentedControl
        value={state.value}
        name="filter"
        onChange={e => setState({ value: e.target.value })}
      >
        <SegmentedControlOption value="all">All</SegmentedControlOption>
        <SegmentedControlOption value="mine">Mine</SegmentedControlOption>
      </SegmentedControl>
    )}
  </State>
</Playground>
